Percentage of Farming employment by county

employment$Total = rowSums(employment[, 2:10])

farming = data.frame(
  region = employment$STCOU,
  value = employment$Farming / employment$Total
)

county_choropleth(farming, title = "2000 Census Farming Employment", legend = "Count")
