Eigen Values and Statistical Distance

## Generate some data
library(mvtnorm)

mu = rep(10, 6)
cov = matrix(nrow = 6, byrow = TRUE,
  c(1, 0, 1, 0, 1, 1,
    0, 2, 1, 2, 1, 1,
    1, 1, 2, 1, 1, 0,
    0, 2, 1, 3, 2, 1,
    1, 1, 1, 2, 4, 1,
    1, 1, 0, 1, 1, 10)
  )

set.seed(1000)
X = rmvnorm(100, mu, cov)
